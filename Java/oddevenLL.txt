


class Solution {
    public ListNode oddEvenList(ListNode head) {
        if(head==null){
            return null;
        }
        ListNode odd=head;
        ListNode even=head.next;
        ListNode evenhead=even;
        
        while(true){
            if(odd==null||even==null||even.next==null){
                odd.next=evenhead;
                break;
            }
            odd.next=even.next;
            odd=even.next;
            
            if(odd.next==null){
                even.next=null;
                odd.next=evenhead;
                break;
            }
            even.next=odd.next;
            even=odd.next;
        }
        return head;
    }
}
