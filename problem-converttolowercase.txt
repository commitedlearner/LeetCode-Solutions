class Solution {
    public String toLowerCase(String str) {
         StringBuilder newstr = new StringBuilder();
        for(int i = 0; i < str.length(); i++) {
            char c = str.charAt(i);
            if (c >= 'A' && c <= 'Z') {
                c += 32;
            }
            newstr.append(c);
        }
        return newstr.toString();
    }
}
