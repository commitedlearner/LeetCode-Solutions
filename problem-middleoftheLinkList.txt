class Solution {
    public ListNode middleNode(ListNode head) {
        ListNode slow=head;
        ListNode fast=head;
        int size=0;
        while( head!=null){
            size++;
            head=head.next;
        }
        
        while(fast.next!=null&&fast.next.next!=null){
            slow=slow.next;
            fast=fast.next.next;
        }
        if(size%2==0){
            return slow.next;
        }else{
            return slow;
        }
       
    }
}
